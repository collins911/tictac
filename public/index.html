const socket = io(); let myPhone = null; let gameId = null; let mySymbol = null; let gameActive = false; let board = Array(9).fill(null); const $ = id => document.getElementById(id); const playBtn = $('play-btn'); const boardWrap = $('board-wrap'); const cells = document.querySelectorAll('.cell'); // ─── AUTH ─── function login() { const phone = $('phone-input').value.trim(); if (!phone) return showToast('Enter phone number', 'fail'); myPhone = phone; socket.emit('auth', { phone }); } socket.on('auth_success', ({ balance }) => { $('wallet-balance').innerText = balance; $('auth-screen').classList.add('hidden'); $('lobby-screen').classList.remove('hidden'); }); // ─── MATCHMAKING ─── function findMatch() { if (!myPhone) return showToast('Not authenticated', 'fail'); playBtn.disabled = true; setStatus('Searching for opponent', 'active', true); socket.emit('find_match'); // UI Watchdog: Re-enable UI if server doesn't find match in 10s setTimeout(() => { if (!gameActive && playBtn.disabled) { playBtn.disabled = false; setStatus('Ready to play', ''); showToast('Search timed out. Try again.', 'warn'); } }, 10000); } socket.on('waiting', () => { setStatus('Waiting for opponent...', 'active', true); }); socket.on('match_found', (data) => { gameId = data.gameId; gameActive = true; $('lobby-screen').classList.add('hidden'); boardWrap.classList.remove('hidden'); }); socket.on('init_game', ({ symbol, turn }) => { mySymbol = symbol; board = Array(9).fill(null); renderBoard(); updateTurnUI(turn); }); // ─── GAMEPLAY ─── cells.forEach((cell, i) => { cell.onclick = () => { if (!gameActive || board[i] || !gameId) return; socket.emit('make_move', { gameId, index: i }); }; }); socket.on('update_board', ({ board: newBoard, turn }) => { board = newBoard; renderBoard(); updateTurnUI(turn); }); function renderBoard() { cells.forEach((cell, i) => { cell.innerText = board[i] || ''; cell.className = 'cell' + (board[i] ? ` taken ${board[i]}` : ''); }); } function updateTurnUI(turn) { if (turn === mySymbol) setStatus("YOUR TURN", "active"); else if (turn) setStatus("OPPONENT'S TURN", ""); } // ─── END ─── socket.on('game_finished', ({ result, winner, prize }) => { gameActive = false; if (result === 'DRAW') showToast(`Draw! Refund: KES ${prize}`, ''); else if (winner === mySymbol) showToast(`WINNER! +KES ${prize}`, 'success'); else showToast('DEFEAT', 'fail'); setTimeout(() => { resetToLobby(); }, 3000); }); socket.on('balance_update', ({ balance }) => { $('wallet-balance').innerText = balance; }); socket.on('error_msg', msg => { showToast(msg, 'fail'); playBtn.disabled = false; setStatus('Ready to play', ''); }); function resetToLobby() { gameId = null; gameActive = false; playBtn.disabled = false; boardWrap.classList.add('hidden'); $('lobby-screen').classList.remove('hidden'); setStatus('Ready to play', ''); } function setStatus(text, className, pulse = false) { const s = $('status'); s.innerText = text; s.className = className + (pulse ? ' pulse' : ''); } function showToast(msg, type) { const t = $('toast'); if (!t) return alert(msg); t.innerText = msg; t.className = `toast show ${type}`; setTimeout(() => t.className = 'toast', 3000); }
